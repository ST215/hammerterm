OpenFront Helper - Usage Examples

Note: This page demonstrates OpenFront Helper usage. To actually use the helper, you must load it on OpenFront.io game page.

Installation

Step 1: Copy the Script

Copy the entire contents of of-helper.v1.js

Step 2: Open Game

Navigate to OpenFront.io and start or join a game

Step 3: Open Console

Press F12 or Ctrl+Shift+J to open browser console

Step 4: Paste & Run

Paste the script and press Enter

Step 5: Verify

You should see console messages and a HUD overlay appear in top-right corner

Basic API Usage

Get Current State

// Get current game state
const state = OFHelper.getState();

console.log('Tick:', state.tick);
console.log('My tiles:', state.my.tiles);
console.log('My troops:', state.my.troops);
console.log('My gold:', state.my.gold);

Listen to Tick Events

// Log troops every tick
OFHelper.on('turn:tick', ({ tick, state }) => {
  console.log(`Tick ${tick}: ${state.my.troops} troops`);
});

Use Selectors

const state = OFHelper.getState();

// Get my player info
const me = OFHelper.selectors.getMyPlayer(state);
console.log('My name:', me.name);

// Get allies
const allies = OFHelper.selectors.getAllies(state);
console.log('Allies:', allies.map(a => a.name));

// Get leaderboard
const top10 = OFHelper.selectors.getLeaderboard(state).slice(0, 10);
top10.forEach((p, i) => {
  console.log(`${i+1}. ${p.name}: ${p.tiles} tiles`);
});

Custom Feature Examples

Example 1: Alliance Monitor

OFHelper.featureRegistry.registerFeature('AllianceMonitor', (api) => {
  const PANEL_ID = 'alliance-monitor';

  function updateAlliances({ state }) {
    const allies = api.selectors.getAllies(state);

    if (allies.length === 0) {
      api.UI.setPanelText(PANEL_ID, 'No allies');
      return;
    }

    const allyInfo = allies.map(a =>
      `${a.name}: ${a.tiles} tiles, ${a.troops} troops`
    ).join('<br>');

    api.UI.setPanelHTML(PANEL_ID, `
      <b>Allies (${allies.length})</b><br>
      ${allyInfo}
    `);
  }

  function init() {
    api.UI.createPanel(PANEL_ID);
    api.on('turn:tick', updateAlliances);
    console.log('[AllianceMonitor] Started');
  }

  function destroy() {
    api.UI.removePanel(PANEL_ID);
    console.log('[AllianceMonitor] Stopped');
  }

  return { init, destroy };
});

Example 3: Troop Capacity Warning

OFHelper.featureRegistry.registerFeature('TroopWarning', (api) => {
  const PANEL_ID = 'troop-warning';
  const THRESHOLD = 0.9;
  let lastWarnTick = -1;

  function checkCapacity({ tick, state }) {
    const ratio = state.my.troops / state.my.troopCap;

    if (ratio >= THRESHOLD) {
      const percentage = Math.floor(ratio * 100);
      api.UI.setPanelHTML(PANEL_ID, `
        <span style="color: #ff0">
          ⚠️ Troops at ${percentage}%!
        </span>
      `);

      // Console warning once per 100 ticks
      if (tick - lastWarnTick >= 100) {
        console.warn('[TroopWarning] Approaching troop capacity!');
        lastWarnTick = tick;
      }
    } else {
      api.UI.setPanelHTML(PANEL_ID, `
        <span style="color: #0f0">
          ✓ Troops OK (${Math.floor(ratio * 100)}%)
        </span>
      `);
    }
  }

  function init() {
    api.UI.createPanel(PANEL_ID);
    api.on('turn:tick', checkCapacity);
  }

  function destroy() {
    api.UI.removePanel(PANEL_ID);
  }

  return { init, destroy };
});

Example 4: Leaderboard Panel

OFHelper.featureRegistry.registerFeature('Leaderboard', (api) => {
  const PANEL_ID = 'leaderboard';
  const TOP_N = 5;

  function updateLeaderboard({ state }) {
    const top = api.selectors.getLeaderboard(state).slice(0, TOP_N);

    const html = top.map((player, index) => {
      const isMe = player.smallID === state.myPlayerId;
      const color = isMe ? '#ff0' : '#fff';
      return `<span style="color: ${color}">${index + 1}. ${player.name}: ${player.tiles}</span>`;
    }).join('<br>');

    api.UI.setPanelHTML(PANEL_ID, `
      <b>Top ${TOP_N}</b><br>
      ${html}
    `);
  }

  function init() {
    api.UI.createPanel(PANEL_ID);
    api.on('turn:tick', updateLeaderboard);
  }

  function destroy() {
    api.UI.removePanel(PANEL_ID);
  }

  return { init, destroy };
});

Advanced Examples

Example: Territory Percentage Tracker

OFHelper.featureRegistry.registerFeature('TerritoryTracker', (api) => {
  const PANEL_ID = 'territory';
  const MAP_SIZE = 10000; // 100x100 tiles

  function updateTerritory({ state }) {
    const myPercent = (state.my.tiles / MAP_SIZE * 100).toFixed(2);
    const alive = api.selectors.getAlivePlayers(state);
    const avgPercent = (MAP_SIZE / alive.length / MAP_SIZE * 100).toFixed(2);

    api.UI.setPanelHTML(PANEL_ID, `
      <b>Territory</b><br>
      Me: ${myPercent}%<br>
      Avg: ${avgPercent}%<br>
      Players: ${alive.length}
    `);
  }

  function init() {
    api.UI.createPanel(PANEL_ID);
    api.on('turn:tick', updateTerritory);
  }

  function destroy() {
    api.UI.removePanel(PANEL_ID);
  }

  return { init, destroy };
});

Example: Interactive Feature with Toggle

OFHelper.featureRegistry.registerFeature('ToggleablePanel', (api) => {
  const PANEL_ID = 'toggle-panel';
  let enabled = true;

  function toggleDisplay() {
    enabled = !enabled;
    api.UI.setPanelVisible(PANEL_ID, enabled);
    console.log('[ToggleablePanel]', enabled ? 'Shown' : 'Hidden');
  }

  // Expose toggle to global scope
  window.togglePanel = toggleDisplay;

  function updatePanel({ state }) {
    if (!enabled) return;
    api.UI.setPanelHTML(PANEL_ID, `
      <b>Interactive Panel</b><br>
      Tick: ${state.tick}<br>
      <small>Type 'togglePanel()' in console to toggle</small>
    `);
  }

  function init() {
    api.UI.createPanel(PANEL_ID, { interactive: true });
    api.on('turn:tick', updatePanel);
    console.log('[ToggleablePanel] Use togglePanel() to show/hide');
  }

  function destroy() {
    api.UI.removePanel(PANEL_ID);
    delete window.togglePanel;
  }

  return { init, destroy };
});

Debugging & Development

Debug: Log All Events

// Listen to all events for debugging
['wire:gameUpdate', 'state:updated', 'turn:tick'].forEach(event => {
  OFHelper.on(event, (payload) => {
    console.log(`[${event}]`, payload);
  });
});

Debug: State Snapshots

// Take periodic state snapshots
let snapshots = [];

OFHelper.on('turn:tick', ({ tick, state }) => {
  if (tick % 100 === 0) {  // Every 100 ticks (10 seconds)
    snapshots.push({
      tick,
      tiles: state.my.tiles,
      troops: state.my.troops,
      gold: state.my.gold
    });
    console.log('Snapshot taken at tick', tick);
  }
});

// View snapshots
console.table(snapshots);
Warning: These examples are for educational purposes. When creating automation features, be mindful of game rules and fair play. The helper is designed primarily for information display, not automation.

Troubleshooting

Common Issues

HUD Not Showing

  • Make sure you're on the game page (not lobby)
  • Check console for errors
  • Verify script loaded: console.log(window.OFHelper)

No Updates Happening

  • Ensure game is running (not paused)
  • Check if events are firing: OFHelper.on('turn:tick', () => console.log('tick'))
  • Verify Wiretap initialized: check for "[Wiretap] Initialized" message

Script Errors

  • Clear console and reload page
  • Paste script again
  • Check browser compatibility (need ES6 support)
